1. Генерация ключа
    openssl genrsa -des3 -out server.key.secure 2048
2. Рашифрование ключа
    openssl rsa -in server.key.secure -out server.key
3. Создание файла сертификата
    openssl req -new -key server.key -out server.csr
4. Самоподписание сертификата
    openssl x509 -req -days 9365 -in server.csr -signkey server.key -out server.crt
